<script>
    const currentPath = window.location.pathname;
    $('.buttons a').attr('target', '_self');
    // Define page mappings - easy to add new pages here
    const pages = [
        {
            // Homepage: exact match only
            condition: currentPath.endsWith('/index.html'),
            selector: '.home-nav',
            replacement: '<b>Home </b>|'
        },
        {
            // Projects: folder or filename match
            condition: currentPath.endsWith('projects.html'),
            selector: '.projects-nav',
            replacement: '| <b>Projects</b>'
        },
        {
            // Academia: folder or filename match
            condition: currentPath.endsWith('academia.html'),
            selector: '.academia-nav',
            replacement: '| <b>Academia</b>'
        },
        {
            // Links: folder or filename match
            condition: currentPath.endsWith('links.html'),
            selector: '.links-nav',
            replacement: '| <b>More links</b>'
        }
    ];

    // Find and replace current page's navigation with blank space
    pages.forEach(page => {
        if (page.condition) {
            $(page.selector).html(page.replacement);
        }
    });
</script>
